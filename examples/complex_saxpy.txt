# Complex SAXPY with multiple scalars
# Z = a*X + b*Y + c

DATA X = [1, 2, 3, 4, 5, 6, 7, 8]
DATA Y = [10, 20, 30, 40, 50, 60, 70, 80]
SCALAR a = 2
SCALAR b = 3
SCALAR c = 5

VLOAD V1, X
VLOAD V2, Y

# a * X
VMULS V3, V1, a

# b * Y
VMULS V4, V2, b

# a*X + b*Y
VADD V5, V3, V4

# a*X + b*Y + c
VADDS V6, V5, c

VSTORE Z, V6
