# Vector Mask Operation Example
# Tạo mask cho các phần tử > 5, sau đó chỉ cộng các phần tử được mask

DATA A = [1, 2, 3, 4, 5, 6, 7, 8]
DATA B = [10, 20, 30, 40, 50, 60, 70, 80]

VLOAD V1, A
VLOAD V2, B

# Tạo mask: các phần tử > 5
VMASK V9, V1, gt

# Chỉ cộng các phần tử có mask=1
VADD_MASKED V3, V1, V2, V9

VSTORE C, V3
